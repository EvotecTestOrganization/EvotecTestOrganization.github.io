#------------------------------Script#1-------Printsvrback.ps1-----------------------------------------------------------------------
#---------------Powershell script to backup site server printer config---------------------------------------------------------------
#----Created by Mano25---------------------------------------------------------------------------------------------------------------
#----Text file with site server names or IP address.
$a = Get-Content "C:\Scripts\allservers.txt"
foreach ($i in $a)
{ 
#------------Create c:\printserver_backup folder
md \\$i\c$\PrintServer_Backup; remove-item -path \\$i\c$\PrintServer_Backup\$i.bak -force; cmd /c "\\$i\c$\windows\system32\spool\tools\printbrm.exe -s \\$i -b -f \\$i\c$\PrintServer_Backup\$i.bak" 

}

#---------------------------------EOF------------------------------------------------------------------------------------------------






--------------------------------Script#2 to copy config file to HO/off-site server---------------------------------------------------
#----Text file with site server names or IP address----------------------------------------------------------------------------------
$a = Get-Content "C:\Scripts\allservers.txt"

foreach ($i in $a)
   { xcopy /E /Y \\$i\c$\PrintServer_Backup\*.bak \\HOSERVER1\D$\PrintServer_Backup\ >> backup2sim.log}
#---------------------------------EOF------------------------------------------------------------------------------------------------
